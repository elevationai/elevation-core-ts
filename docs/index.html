@eai/elevation-core-ts

Comprehensive TypeScript SDK for Elevated Platform Services.
Real-time reporting, bidirectional communication, and remote logging for touchpoint devices.

JSR Package TypeScript Enterprise Ready
Get Started

Powerful Features

Real-time IOT

WebSocket-based bidirectional communication for instant device control, command handling, and real-time updates with automatic reconnection.

Event Analytics

Comprehensive event system with 200+ predefined codes, custom events, automatic metadata enrichment, and intelligent debouncing.

Centralized Logging

Multiple log levels (INFO, DELAYED, ERROR, CRITICAL) with automatic device association and configurable debouncing.

Device Enrollment

Automated device registration with location-based configuration, terminal association, and specification management.

Configuration Management

Dynamic configuration with hierarchical overrides (global → location → device) and real-time updates.

Content Management

Multi-language content with versioning, scheduled delivery, draft/published workflows, and automatic language fallback.

import { events, EventCode, StatusCode } from '@jsr/eai__elevation-core-ts';

// Configure events service
const coreInfo = {
  token: 'your-access-token',
  serviceEndpoint: 'https://api.example.com'
};

events.config(coreInfo);

// Set default values
events.setDefaults({
  eventType: EventType.CHECKIN_KIOSK,
  eventMode: EventMode.CUSS,
  ownerID: 'device-guid'
});

// Send events with helper methods
await events.success({ 
  eventCode: EventCode.BAGTAG_PRINT,
  eventData: { PNR: 'ABC123', airline: 'AA' }
});

await events.failure({ 
  eventCode: EventCode.PRINTER_ERROR 
});

await events.critical({ 
  eventCode: EventCode.OUT_OF_SERVICE 
});

import { iot, Commands } from '@jsr/eai__elevation-core-ts';

// Configure IOT service
const coreInfo = {
  token: 'your-access-token',
  serviceEndpoint: 'https://api.example.com',
  iotEndpoint: 'https://iot.example.com',
  fingerPrint: 'unique-device-id'
};

const iotInfo = {
  appName: 'CheckinApp',
  appVersion: '1.0.0'
};

iot.config(coreInfo, iotInfo);

// Handle connection events
iot.onConnected.subscribe(() => {
  console.log('Connected to IOT service');
});

iot.onCommand.subscribe((command: Commands) => {
  if (command.restart) {
    console.log('Restart command received');
  }
});

// Update device service state
await iot.inService('device-123', true, 'Operational');
import { 
  enrollment, CoreInfo, DeviceUpdate, 
  DeviceLocation, Specification, DeviceInfo
} from '@jsr/eai__elevation-core-ts';

// Configure enrollment service
const coreInfo: CoreInfo = {
  token: 'your-access-token',
  serviceEndpoint: 'https://api.example.com',
  fingerPrint: 'unique-device-id'
};

enrollment.config(coreInfo);

// Start enrollment process
Promise.all([
  enrollment.start(),
  enrollment.getLocations(),
  enrollment.getSpecification()
]).then(async ([
  device: DeviceUpdate, 
  locations: DeviceLocation[], 
  specs: Specification[]
]) => {
  // Create device object for enrollment
  const devObj: DeviceInfo = {
    label: 'KIOSK001', // unique name
    device: device,
    location: locations[0],
    terminal: locations[0]?.terminals[0],
    specification: specs[0]
  };

  // Complete enrollment
  await enrollment.enrollDevice(devObj);
});

import { cms, CoreInfo } from '@jsr/eai__elevation-core-ts';
import { firstValueFrom } from 'rxjs';

// Configure CMS with version and draft mode
const coreInfo: CoreInfo = {
  token: 'your-access-token',
  serviceEndpoint: 'https://api.example.com',
  version: 'v2.0', // Optional: use specific content version
  isDraft: false // Optional: use published content (default)
};

cms.config(coreInfo);

// Get localized strings (with auto-fallback to en-US)
const welcomeMsg = await cms.getString('welcome_message', 'en-US');
const spanishMsg = await cms.getString('welcome_message', 'es-ES');

// Get multiple localized strings at once
const keys = ['welcome_message', 'goodbye_message', 'error_message'];
const messages = await cms.getKeys(keys, 'en-US');

// Get all available language codes
const languages = await cms.getLangs();
console.log(languages); // ["en-US", "es-ES", "fr-FR"]

// Get configuration objects (automatically parsed JSON)
const themeConfig = await cms.getConfig('app_theme_config', 'en-US');
console.log(themeConfig.primaryColor); // "#007bff"

// Pre-load all CMS content
await firstValueFrom(cms.loadAllStrings());

// Cache management
const stats = cms.getCacheStats();
console.log(`Cache has ${stats.size} entries`);

// Clear cache and force fresh data
cms.clearCache();
const freshMsg = await cms.getString('welcome_message', 'en-US', false);
import { touchPoint } from '@jsr/eai__elevation-core-ts';

// Configure touchpoint service
const coreInfo = {
  token: 'your-access-token',
  serviceEndpoint: 'https://api.example.com',
  fingerPrint: 'unique-device-id'
};

touchPoint.config(coreInfo);

// Get device information
const deviceInfo = await touchPoint.getInfo();
if (deviceInfo) {
  console.log('Device:', deviceInfo.label);
  console.log('Location:', deviceInfo.location);
}

// Update service state
await touchPoint.inService(true, 'Device operational');

// Take out of service
await touchPoint.inService(false, 'Maintenance required');
